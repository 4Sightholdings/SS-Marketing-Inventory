function upload(){ let file = fileInput.files[0]; if(!file){ alert("Please select a file"); return; } let reader = new FileReader(); reader.onload = function(e){ try{ let wb = XLSX.read(e.target.result,{type:'array'}); let sheet = wb.Sheets[wb.SheetNames[0]]; // ✅ FORCE HEADER ROW HANDLING let json = XLSX.utils.sheet_to_json(sheet, { defval: "", // prevent empty values breaking things raw: false // better date handling }); console.log("RAW DATA:", json); if(!json || json.length === 0){ alert("⚠️ File loaded but no readable data found.\nCheck column names."); return; } // ✅ CLEAN COLUMN NAMES data = json.map(row => { let clean = {}; Object.keys(row).forEach(key => { clean[key.trim()] = row[key]; }); return clean; }); console.log("CLEAN DATA:", data); localStorage.setItem("clickData", JSON.stringify(data)); render(); } catch(err){ console.error(err); alert("❌ Error reading Excel file"); } }; reader.readAsArrayBuffer(file); }
